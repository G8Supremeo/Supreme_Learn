useLocalStorage — Persistence

Synchronises React state with browser storage using a lazy initialiser to prevent expensive Disk I/O on re-renders.


export function useLocalStorage(key, initialValue) {
  const [storedValue, setStoredValue] = useState(() => {
    try {
      const item = window.localStorage.getItem(key);
      return item ? JSON.parse(item) : initialValue;
    } catch { return initialValue; }
  });

  const setValue = (value) => {
    const valueToStore = value instanceof Function ? value(storedValue) : value;
    setStoredValue(valueToStore);
    window.localStorage.setItem(key, JSON.stringify(valueToStore));
  };
  return [storedValue, setValue];
}

useNotification — Lifecycle

Uses useCallback to ensure function stability across the component tree.


export function useNotification() {
  const [notification, setNotification] = useState(null);

  const showNotification = useCallback((message, type = 'success', duration = 3000) => {
    setNotification({ message, type });
    setTimeout(() => setNotification(null), duration);
  }, []);

  return { notification, showNotification };
}

ExamWidget Architecture

A sophisticated state machine for managing large-scale assessments.

  • Result Computation: 70% pass threshold filter.

  });
